多态真值
  二态真值的值集 := {真,假,}
  三态真值的值集 := {待,真,假,}
  四态真值的值集 := {错,待,真,假,}
    {error/exception,yet/unknown/todo/fail,true/yes, false/no}
  前提不成立 <==> 出错
    比如: 定义不存在
  前提成立但尚未求出结果 <==> 待办
    比如: 整数素性判定，整数分解，图同构判定，结同构判定，命题真伪判定
  特例定义 =[def]= [前提 => 定义语句]
  四态命题 =[def]= [二态前提 => 三态命题]
异常求值:
  异常求值 =[def]= [立即求值前提 => 断续求值]
  立即值囗值集 =[def]= {*立即值}
  断续值囗值集 =[def]= {待,*立即值}
  异常值囗值集 =[def]= {错,待,*立即值}
